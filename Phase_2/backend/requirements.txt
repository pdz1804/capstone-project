# ======================== System Dependencies Required ========================
# - FFmpeg: Required for video/audio processing
#   Windows: Download from https://ffmpeg.org/download.html
#   Linux: sudo apt-get install ffmpeg
#   Mac: brew install ffmpeg
#
# - Tesseract OCR: Required for text extraction
#   Windows: https://github.com/UB-Mannheim/tesseract/wiki
#   Linux: sudo apt-get install tesseract-ocr
#   Mac: brew install tesseract
#
# - Poppler (for PDF to image): Required for multi-modal retrieval
#   Windows: Download from http://blog.alivate.com.au/poppler-windows/
#   Linux: sudo apt-get install poppler-utils
#   Mac: brew install poppler
#
# - LibreOffice: Required for DOCX/PPTX → PDF conversion
#   Windows: Download from https://www.libreoffice.org/download/
#   Linux: sudo apt-get install libreoffice
#   Mac: brew install libreoffice

# ======================== Installation Instructions ========================
# 1. Create virtual environment:
#    python -m venv venv
#
# 2. Activate virtual environment:
#    Windows: venv\Scripts\activate
#    Linux/Mac: source venv/bin/activate
#
# 3. Install requirements:
#    pip install -r requirements.txt
#
# 4. For GPU support, install PyTorch with CUDA (here it is 11.8):
#    pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
#
# 5. For GPU FAISS support (optional):
#    pip uninstall faiss-cpu
#    pip install faiss-gpu
#
# 6. Download models (first run will auto-download):
#    python -c "import whisper; whisper.load_model('base')"
#    python -c "from sentence_transformers import SentenceTransformer; SentenceTransformer('all-MiniLM-L6-v2')"
